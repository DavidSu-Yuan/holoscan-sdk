cmake_minimum_required(VERSION 3.20)
project(holoscan_native_operator CXX)

# Finds the package holoscan
find_package(holoscan REQUIRED)

add_executable(ping
  ping.cpp
)

target_link_libraries(ping
  PRIVATE
  holoscan::holoscan
)

# Copy config file to the build tree
add_custom_target(ping_yaml
    COMMAND ${CMAKE_COMMAND} -E copy "${CMAKE_SOURCE_DIR}/app_config.yaml" ${CMAKE_CURRENT_BINARY_DIR}
)
add_dependencies(ping ping_yaml)
